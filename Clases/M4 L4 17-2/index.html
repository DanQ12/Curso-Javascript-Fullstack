01 Callback CLASICO
callback.js 
                    // Una funcion que simula una tarea lenta
                    function tareaLenta(nombre, tiempo, callback){
                        console.log(`▶ iniciando: ${nombre}`)

                        setTimeout(function{
                            callback(`✓ ${nombre} completada`)
                        }, timepo)
                    }

                    // Usando el callback
                    tareaLenta("Descargar datos", 2000, function(res) { 
                        console.log(res)
                        // Otro callback dentro → "Callback Hell"
                        tareaLenta("Procesar datos", 1000, function(res2){
                            console.log(res2)
                        })
                    })

                    console.log("⚡ Codigo sincrono sigue corriendo...")

⚠ Callback Hell Anidar callbacks dentro de callbacks hace el codigo dificil de leer y mantener. Por eso sugiero las Promesas.
02 Promise (Promesa) MODERNO
promise.js 
                    function tareaPromesa(nombre, tiempo, exito){
                        return new Promise((resolve, reject)=>{
                            console.log(`▶ Iniciando: ${nombre}`)
                            setTimeout(() => {
                                if (exito){
                                    resolve(`✓ ${nombre} completada`)
                                }else {
                                    reject(`✗ ${nombre} fallo`)
                                }
                            }, tiempo)
                        })
                    }

                    // Encadenado con .then / .catch
                    tareaPromesa("Descargar",1500, true)
                        .then(res=>{
                            console.log(res)
                            return tareaPromesa("Procesar", 1000, true)
                        })
                        .then(res => console.log(res))
                        .catch(err => console.log(err))

                        console.log("⚡ Codigo sincrono continua ...")
CONSOLA
// Prueba EXITO o ERROR
Estados de una Promise: pending → en espera · fulfilled → resuelto · rejected → rechazado
03 Async / Await RECOMENDADO
async-await.js 
                    function esperar(ms){
                        return new Promise(res => setTimeout(res, ms))
                    }

                    // async hace que la funcion retorne una Promise
                    async function procesarDatos(){
                        try{
                            console.log("▶ Paso 1: Conectando...")
                            await esperar(1200)
                            console.log("✓ Paso 1: Conexion OK")

                            console.log("▶ Paso 2: Descargando...")
                            await esperar(1500)
                            console.log("✓ Paso 2: Descarga OK")

                            console.log("▶ Paso 3: Procesando...")
                            await esperar(800)
                            console.log("✓ Paso 3: Proceso completado 🎉")
                        } catch (error){
                            console.log(`✗ Error: ${error}`)
                        }
                    }

                    procesarDatos()
                    console.log("⚡ Esto se imprime antes de los pasos")
CONSOLA
// Presionar EJECUTAR para ver la salida...
await pausa la ejecucion dentro de la funcion async, pero no bloquea el resto del programa, es como usar .then() pero con sintaxis de codigo normal.
